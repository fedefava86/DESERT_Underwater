165  tcl.resultf("%d", getWrongNodeDataSent());
166  return TCL_OK;
167  } else if (strcasecmp(argv[1], "getProbeReceived") == 0) {
168  tcl.resultf("%d", getProbeRx());
169  return TCL_OK;
170  } else if (strcasecmp(argv[1], "getPollSent") == 0) {
171  tcl.resultf("%d", getPollSent());
172  return TCL_OK;
173  } else if (strcasecmp(argv[1], "getDroppedProbePkts") == 0) {
174  tcl.resultf("%d", getDroppedProbePkts());
175  return TCL_OK;
176  } else if (strcasecmp(argv[1], "getDroppedProbeWrongState") == 0) {

278  if (sea_trial_ && print_stats_)

280  << "::Uwpolling_AUV(" << addr << ")::TX_TRIGGER_ID_"
281  << TRIGGER_uid << endl;

341 double
342 Uwpolling_AUV::GetDataTimerValue()
343 {
344  UpdateRTT();
345  double timer_value = 0;
346  if (!sea_trial_) {
347  computeTxTime(UWPOLLING_DATA_PKT);
348  timer_value = (N_expected_pkt) * (Tdata + curr_RTT + T_guard);
349  } else {
350  double Tdata = (max_payload * 8) / modem_data_bit_rate;
351  timer_value = (N_expected_pkt) * (Tdata + curr_RTT + T_guard);
352  }
353  if (timer_value <= 0) {
354  cout << "Uwpolling_AUV(" << addr
355  << ")::WARNING::Data Timer Value <= 0.Setting HARDCODED = 20 sec"
356  << std::endl;
357  timer_value = 20;
358  }
359  return timer_value;

377 void
378 Uwpolling_AUV::Phy2MacEndRx(Packet *p)
379 {
380  hdr_cmn *cmh = hdr_cmn::access(p);
381  hdr_mac *mach = HDR_MAC(p);
382  hdr_MPhy *ph = hdr_MPhy::access(p);
383  int dest_mac = mach->macDA();
384  double gen_time = ph->txtime;
385  double received_time = ph->rxtime;
386  double diff_time = received_time - gen_time;
387 
388  if (cmh->error()) {
389  if (cmh->ptype_ == PT_POLL) {
390  std::cout << NOW << "Uwpolling_AUV(" << addr
391  << ")::PHY2MACENDRX_DROP_POLL" << endl;
392  } else if (cmh->ptype_ == PT_PROBE) {
393  if (debug_)
394  std::cout << NOW << "Uwpolling_AUV(" << addr
395  << ")::PHY2MACENDRX_DROP_PROBE_FROM_" << mach->macSA()
396  << endl;
397  if (sea_trial_ && print_stats_)
398  out_file_stats << left << NOW << "Uwpolling_AUV(" << addr
399  << ")::PHY2MACENDRX::DROP_PROBE_FROM_NODE_"
400  << mach->macSA() << endl;
401  incrDroppedProbePkts();
402  } else if (cmh->ptype_ == PT_TRIGGER) {
403  std::cerr << NOW << "Uwpolling_AUV(" << addr
404  << ")::PHY2MACENDRX_DROP_TRIGGER" << endl;
405  } else {
406  if (debug_)
407  std::cout << NOW << "Uwpolling_AUV(" << addr
408  << ")::PHY2MACENDRX::DROP_DATA_FROM_NODE_"
409  << mach->macSA() << endl;
410  if (sea_trial_ && print_stats_)
411  out_file_stats << left << NOW << "Uwpolling_AUV(" << addr
412  << ")::PHY2MACENDRX::DROP_DATA_FROM_NODE_"
413  << mach->macSA() << endl;
414  incrErrorPktsRx();
415  }
416  refreshReason(UWPOLLING_AUV_REASON_PACKET_ERROR);
417  drop(p, 1, UWPOLLING_AUV_DROP_REASON_ERROR);
418  } else {
419  if ((dest_mac == addr) || (dest_mac == (int) MAC_BROADCAST)) {
420  if (cmh->ptype_ == PT_PROBE) {
421  incrCtrlPktsRx();
422  refreshReason(UWPOLLING_AUV_REASON_PROBE_RECEIVED);
423  curr_probe_packet = p->copy();
424  Packet::free(p);
425  hdr_PROBE *prh = hdr_PROBE::access(curr_probe_packet);
426  if (debug_)
427  std::cout << NOW << "Uwpolling_AUV(" << addr
428  << ")::PHY2MACENDRX::PROBE pkt::N_pkt="
429  << prh->n_pkts_
430  << "::Backoff_time=" << prh->backoff_time()
431  << "::Id_nodo=" << prh->id_node() << std::endl;
432  probe_rtt = diff_time;
433  if (begin) {
434  begin = false;
435  initial_time = NOW;
436  }
437  stateRxProbe();
438  } else {
439  if ((cmh->ptype_ != PT_POLL) && (cmh->ptype_ != PT_PROBE) &&
440  (cmh->ptype_ != PT_TRIGGER)) {
441  refreshReason(UWPOLLING_AUV_REASON_DATA_RX);
442  curr_data_packet = p->copy();
443  Packet::free(p);
444  pkt_time = NOW;
445  stateRxData();
446  }
447  }
448  } else {
449  if (debug_)
450  std::cout << NOW << "Uwpolling_AUV(" << addr
451  << ")::PHY2MACENDRX::DROP_DATA_WRONG_DEST_"
452  << mach->macDA() << endl;
453  incrXCtrlPktsRx();
454  if (sea_trial_ && print_stats_)
455  out_file_stats << left << "[" << getEpoch() << "]::" << NOW
456  << "::Uwpolling_AUV(" << addr
457  << ")::PHY2MACENDRX::DROP_DATA_WRONG_DEST_"
458  << mach->macDA() << endl;
459  refreshReason(UWPOLLING_AUV_REASON_PACKET_ERROR);
460  drop(p, 1, UWPOLLING_AUV_DROP_REASON_WRONG_RECEIVER);
461  }
462  }

465 void
466 Uwpolling_AUV::computeTxTime(UWPOLLING_PKT_TYPE pkt)
467 {
468  Packet *fuzzy_pkt;
469  fuzzy_pkt = Packet::alloc();
470  switch (pkt) {
471  case (UWPOLLING_POLL_PKT): {
472  hdr_cmn *cmh = HDR_CMN(fuzzy_pkt);
473  hdr_mac *mach = HDR_MAC(fuzzy_pkt);
474  mach->macSA() = addr;
475  cmh->size() = sizeof(id_poll) * polling_index;
476  Tpoll = Mac2PhyTxDuration(fuzzy_pkt);
477  } break;
478  case (UWPOLLING_PROBE_PKT): {
479  hdr_cmn *cmh = HDR_CMN(fuzzy_pkt);
480  hdr_mac *mach = HDR_MAC(fuzzy_pkt);
481  mach->macSA() = addr;
482  cmh->size() = sizeof(hdr_PROBE);
483  Tprobe = Mac2PhyTxDuration(fuzzy_pkt);
484  } break;
485  case (UWPOLLING_TRIGGER_PKT): {
486  hdr_cmn *cmh = HDR_CMN(fuzzy_pkt);
487  cmh->size() = sizeof(hdr_TRIGGER);
488  hdr_mac *mach = HDR_MAC(fuzzy_pkt);
489  mach->macSA() = addr;
490  // cmh->size() = 10;
491  Ttrigger = Mac2PhyTxDuration(fuzzy_pkt);
492  } break;
493  case (UWPOLLING_DATA_PKT): {
494  hdr_cmn *cmh = HDR_CMN(fuzzy_pkt);
495  hdr_mac *mach = HDR_MAC(fuzzy_pkt);
496  mach->macDA() = addr;
497  cmh->size() = max_payload;
498  Tdata = Mac2PhyTxDuration(fuzzy_pkt);
499  }
500  }
501  Packet::free(fuzzy_pkt);
502 }

513 void
514 Uwpolling_AUV::stateRxData()
515 {
516  refreshState(UWPOLLING_AUV_STATUS_RX_DATA);
517  if (RxDataEnabled) {
518  hdr_mac *mach = HDR_MAC(curr_data_packet);
519  hdr_uwcbr *cbrh = HDR_UWCBR(curr_data_packet);
520  if (sea_trial_ && print_stats_)
521  out_file_stats << left << "[" << getEpoch() << "]::" << NOW
522  << "::Uwpolling_AUV(" << addr
523  << ")::STATE_RX_DATA::RX_DATA_ID_" << cbrh->sn_
524  << "_FROM_NODE_" << mach->macSA() << endl;
525  if (mach->macSA() == curr_polled_node_address) {
526  if (debug_)
527  std::cout << NOW << "Uwpolling_AUV(" << addr
528  << ")::STATE_RX_DATA::RX_DATA_ID_" << cbrh->sn_
529  << "_FROM_NODE_" << mach->macSA() << endl;
530  incrDataPktsRx();
531  packet_index++;
532  sendUp(curr_data_packet);
533  if (packet_index == N_expected_pkt) {
534  refreshReason(UWPOLLING_AUV_REASON_LAST_PACKET_RECEIVED);
535  data_timer.force_cancel();
536  if (sea_trial_) {
537  poll_timer.schedule(DATA_POLL_guard_time_);
538  } else {
539  ChangeNodePolled();
540  }
541  }
542  } else {
543  incrWrongNodeDataSent();
544  if (debug_)
545  std::cout << NOW << "Uwpolling_AUV(" << addr
546  << ")::STATE_RX_DATA::DROP_DATA_WRONG_NODE_"
547  << mach->macSA_ << std::endl;
548  if (sea_trial_ && print_stats_)
549  out_file_stats << left << "[" << getEpoch() << "]::" << NOW
550  << "::Uwpolling_AUV(" << addr
551  << ")::STATE_RX_DATA::DROP_DATA_FROM_NODE_"
552  << mach->macDA() << "_NOT_POLLED" << endl;
553  drop(curr_data_packet, 1, UWPOLLING_AUV_DROP_REASON_WRONG_STATE);
554  }
555  } else {
556  incrXDataPktsRx();
557  hdr_mac *mac = HDR_MAC(curr_data_packet);
558  if (debug_)
559  std::cout << NOW << "Uwpolling_AUV(" << addr
560  << ")::STATE_RX_DATA::DROP_DATA_WRONG_STATE="
561  << status_info[curr_state] << std::endl;
562  if (sea_trial_ && print_stats_)
563  out_file_stats << left << "[" << getEpoch() << "]::" << NOW
564  << "::Uwpolling_AUV(" << addr
565  << ")::STATE_RX_DATA::DROP_DATA_FROM_NODE_"
566  << mac->macDA() << "_NOT_POLLED" << endl;
567  drop(curr_data_packet, 1, UWPOLLING_AUV_DROP_REASON_WRONG_STATE);
568  }

571 void
572 Uwpolling_AUV::ChangeNodePolled()
573 {
574  RxDataEnabled = false;
575  if (debug_)
576  std::cout << NOW << "Uwpolling_AUV(" << addr
577  << ")::CHANGE_NODE_POLLED::" << list_probbed_node[0].id_node
578  << "::" << list_probbed_node[1].id_node
579  << "::" << list_probbed_node[2].id_node
580  << "::" << list_probbed_node[3].id_node << std::endl;
581  if (polling_index > 1) {
582  for (int i = 0; i < polling_index - 1; i++) {
583  list_probbed_node[i] = list_probbed_node[i + 1];
584  }
585  list_probbed_node[polling_index - 1].id_node = (int) NAN;
586  for (int it = polling_index; it < max_polled_node; it++) {
587  list_probbed_node[it].id_node = (int) NAN;
588  }
589  polling_index--;
