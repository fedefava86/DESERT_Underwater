238  virtual void
239  freeze()
240  {
241  assert(timer_status == UWPOLLING_RUNNING);
242  left_duration -= (NOW - start_time);
243  if (left_duration <= 0.0) {
244  left_duration = module->mac2phy_delay_;
245  }
246  force_cancel();
247  timer_status = UWPOLLING_FROZEN;
248  }
249 
254  virtual void
255  unFreeze()
256  {
257  assert(timer_status == UWPOLLING_FROZEN);
258  start_time = NOW;
259  assert(left_duration > 0);
260  sched(left_duration);
261  timer_status = UWPOLLING_RUNNING;
262  }
263 
267  virtual void
268  stop()
269  {
270  timer_status = UWPOLLING_IDLE;
271  force_cancel();
272  }
273 
278  virtual void
279  schedule(double val)
280  {
281  start_time = NOW;
282  left_duration = val;
283  timer_status = UWPOLLING_RUNNING;
284  resched(val);
285  }
